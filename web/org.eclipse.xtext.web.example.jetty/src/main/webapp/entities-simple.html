import java.util.List

/*
 * A simple Entities example.
 */
package test {
	
	// This is my self-made entity
	entity MyEntity {
		myProperty : List
		op myOperation() : String {
			myProperty.head
		}
	}
	
	entity Container {
		entities : List
		op findMatching(String param) : MyEntity {
			entities.findFirst[myOperation == param]
		}
	}
	
}
